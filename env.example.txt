# Monster Slayer - Environment Variables
# Copy this file to .env and fill in your values

# =============================================================================
# SUPABASE CONFIGURATION
# =============================================================================
# Get these from your Supabase project settings > API

VITE_SUPABASE_URL=https://your-project-id.supabase.co
VITE_SUPABASE_ANON_KEY=your-anon-key-here

# =============================================================================
# NOTION INTEGRATION
# =============================================================================
# Get API key from: https://www.notion.so/my-integrations
# Create databases in Notion and share them with your integration
# Get database IDs from the database URL (32 character hex string)

NOTION_API_KEY=secret_your-notion-api-key-here
NOTION_DB_TASKS=your-tasks-database-id-here
NOTION_DB_DOCS=your-docs-database-id-here
NOTION_DB_ASSETS=your-assets-database-id-here
NOTION_DB_COMMITS=your-commits-database-id-here

# How to get database IDs:
# 1. Open your Notion database page
# 2. Copy the URL - it looks like: https://notion.so/workspace/abc123def456...
# 3. The database ID is the 32-character string (without dashes)
# Example: abc123def456789012345678901234567890
